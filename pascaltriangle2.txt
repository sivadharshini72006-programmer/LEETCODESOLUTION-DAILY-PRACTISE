class Solution {
    public List<Integer> getRow(int rowIndex) {
        List<Integer> rows = new ArrayList<>();

        rows.add(1);

        for(int i=1;i<=rowIndex;i++){
            rows.add(0);
            for(int j=i;j>0;j--){
                rows.set(j,rows.get(j)+rows.get(j-1));
            }
        }
        return rows;
        
    }
}
